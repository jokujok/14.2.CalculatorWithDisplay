SELECT ISSUES.SUMMARY, ISSUESLISTS.NAME
FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ID = ISSUESLISTS.ID;


SELECT ISSUES.SUMMARY, USERS.FIRSTNAME, USERS.LASTNAME
FROM ISSUES
JOIN USERS ON ISSUES.ID = USERS.ID;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TO_DO_NUMBER
FROM USERS U JOIN ISSUES I ON U.ID = I.ID
GROUP BY I.ID;
